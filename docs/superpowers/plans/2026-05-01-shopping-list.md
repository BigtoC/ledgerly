# Shopping List Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a draft shopping list anchored in Accounts — users capture future expense items from the transaction form, review and edit them in a dedicated screen, and convert them to real transactions when ready.

**Architecture:** New `shopping_list_items` Drift table (schema v3) with a matching repository, three controllers (`ShoppingListController` for list/undo, `ShoppingListFormController` for capture from the transaction form, `ShoppingListItemController` for create/edit/convert), a `shoppingListCountProvider` that drives a reactive circular FAB shortcut on HomeScreen, and two new screens (`ShoppingListScreen`, `ShoppingListItemScreen`) under `/accounts/shopping-list`. AccountsScreen gains a preview card as its first sliver; its current early-return empty-state guard is refactored to an inline sliver so the card always renders regardless of account count.

**Tech Stack:** Flutter, Drift (SQLite ORM), Riverpod (code-gen with `riverpod_annotation`), Freezed, go_router, flutter_slidable, mocktail (tests), fake_async (timer tests)

```dart
// lib/data/database/daos/shopping_list_dao.dart
import 'package:drift/drift.dart';

import '../app_database.dart';
import '../tables/shopping_list_items_table.dart';

part 'shopping_list_dao.g.dart';

@DriftAccessor(tables: [ShoppingListItems])
class ShoppingListDao extends DatabaseAccessor<AppDatabase>
    with _$ShoppingListDaoMixin {
  ShoppingListDao(super.db);

  Stream<List<ShoppingListItemRow>> watchAll() =>
      (select(shoppingListItems)
            ..orderBy([
              (t) => OrderingTerm(
                expression: t.createdAt,
                mode: OrderingMode.desc,
              ),
            ]))
          .watch();

  Stream<int> watchCount() {
    final countExp = shoppingListItems.id.count();
    final query = selectOnly(shoppingListItems)..addColumns([countExp]);
    return query.watchSingle().map((row) => row.read(countExp) ?? 0);
  }

  Stream<ShoppingListItemRow?> watchById(int id) =>
      (select(shoppingListItems)..where((t) => t.id.equals(id)))
          .watchSingleOrNull();

  Future<ShoppingListItemRow?> findById(int id) =>
      (select(shoppingListItems)..where((t) => t.id.equals(id)))
          .getSingleOrNull();

  Future<int> insertRow(ShoppingListItemsCompanion row) =>
      into(shoppingListItems).insert(row);

  Future<bool> updateRow(ShoppingListItemsCompanion row) =>
      update(shoppingListItems).replace(row);

  Future<int> deleteById(int id) =>
      (delete(shoppingListItems)..where((t) => t.id.equals(id))).go();

  Future<int> insertOnConflictUpdate(ShoppingListItemsCompanion row) =>
      into(shoppingListItems).insertOnConflictUpdate(row);
}
```

```dart
// lib/data/repositories/shopping_list_repository.dart
import '../database/app_database.dart' as drift;
import '../database/daos/shopping_list_dao.dart';
import '../models/shopping_list_item.dart';

class ShoppingListRepositoryException implements Exception {
  const ShoppingListRepositoryException(this.message);
  final String message;
  @override
  String toString() => 'ShoppingListRepositoryException: $message';
}

abstract class ShoppingListRepository {
  Stream<List<ShoppingListItem>> watchAll();
  Stream<int> watchCount();
  Stream<ShoppingListItem?> watchById(int id);
  Future<ShoppingListItem?> getById(int id);
  Future<ShoppingListItem> insert(ShoppingListItem item);
  Future<ShoppingListItem> update(ShoppingListItem item);
  Future<bool> delete(int id);
  Future<void> insertOnConflictUpdate(ShoppingListItem item);
}

final class DriftShoppingListRepository implements ShoppingListRepository {
  DriftShoppingListRepository(this._db, {DateTime Function()? clock})
      : _clock = clock ?? DateTime.now;

  final drift.AppDatabase _db;
  final DateTime Function() _clock;

  ShoppingListDao get _dao => _db.shoppingListDao;

  void _assertCurrencyInvariant(ShoppingListItem item) {
    if (item.draftAmountMinorUnits != null && item.draftCurrencyCode == null) {
      throw const ShoppingListRepositoryException(
        'draftCurrencyCode must be present when draftAmountMinorUnits is non-null',
      );
    }
  }

  @override
  Stream<List<ShoppingListItem>> watchAll() =>
      _dao.watchAll().map((rows) => rows.map(_toDomain).toList());

  @override
  Stream<int> watchCount() => _dao.watchCount();

  @override
  Stream<ShoppingListItem?> watchById(int id) =>
      _dao.watchById(id).map((row) => row == null ? null : _toDomain(row));

  @override
  Future<ShoppingListItem?> getById(int id) async {
    final row = await _dao.findById(id);
    return row == null ? null : _toDomain(row);
  }

  @override
  Future<ShoppingListItem> insert(ShoppingListItem item) async {
    _assertCurrencyInvariant(item);
    final now = _clock();
    final companion = _toCompanion(item, createdAt: now, updatedAt: now);
    final newId = await _dao.insertRow(companion);
    return item.copyWith(id: newId, createdAt: now, updatedAt: now);
  }

  @override
  Future<ShoppingListItem> update(ShoppingListItem item) async {
    _assertCurrencyInvariant(item);
    final now = _clock();
    final companion = _toCompanion(item, createdAt: item.createdAt, updatedAt: now);
    await _dao.updateRow(companion);
    return item.copyWith(updatedAt: now);
  }

  @override
  Future<bool> delete(int id) async {
    final removed = await _dao.deleteById(id);
    return removed > 0;
  }

  @override
  Future<void> insertOnConflictUpdate(ShoppingListItem item) async {
    final companion = _toCompanion(
      item,
      createdAt: item.createdAt,
      updatedAt: item.updatedAt,
    );
    await _dao.insertOnConflictUpdate(companion);
  }

  ShoppingListItem _toDomain(drift.ShoppingListItemRow row) => ShoppingListItem(
    id: row.id,
    categoryId: row.categoryId,
    accountId: row.accountId,
    memo: row.memo,
    draftAmountMinorUnits: row.draftAmountMinorUnits,
    draftCurrencyCode: row.draftCurrencyCode,
    createdAt: row.createdAt,
    updatedAt: row.updatedAt,
  );

  drift.ShoppingListItemsCompanion _toCompanion(
    ShoppingListItem item, {
    required DateTime createdAt,
    required DateTime updatedAt,
  }) =>
      drift.ShoppingListItemsCompanion(
        id: item.id == 0
            ? const drift.Value.absent()
            : drift.Value(item.id),
        categoryId: drift.Value(item.categoryId),
        accountId: drift.Value(item.accountId),
        memo: item.memo == null
            ? const drift.Value.absent()
            : drift.Value(item.memo),
        draftAmountMinorUnits: item.draftAmountMinorUnits == null
            ? const drift.Value.absent()
            : drift.Value(item.draftAmountMinorUnits),
        draftCurrencyCode: item.draftCurrencyCode == null
            ? const drift.Value.absent()
            : drift.Value(item.draftCurrencyCode),
        createdAt: drift.Value(createdAt),
        updatedAt: drift.Value(updatedAt),
      );
}
```

```dart
// lib/features/shopping_list/shopping_list_controller.dart
import 'dart:async';

import 'package:riverpod_annotation/riverpod_annotation.dart';

import '../../app/providers/repository_providers.dart';
import '../../core/constants.dart';
import '../../data/models/shopping_list_item.dart';
import 'shopping_list_state.dart';

part 'shopping_list_controller.g.dart';

typedef ShoppingListEffectListener = void Function(ShoppingListEffect);

sealed class ShoppingListEffect { const ShoppingListEffect(); }
final class ShoppingListDeleteFailedEffect extends ShoppingListEffect {
  const ShoppingListDeleteFailedEffect(this.error);
  final Object error;
}

@Riverpod(keepAlive: true, dependencies: [shoppingListRepository])
class ShoppingListController extends _$ShoppingListController {
  List<ShoppingListItem> _allItems = [];
  ShoppingListItem? _pendingDelete;
  final Set<int> _committedDeleteIds = {};
  Timer? _undoTimer;
  ShoppingListEffectListener? _effectListener;

  @override
  Stream<ShoppingListState> build() {
    final repo = ref.watch(shoppingListRepositoryProvider);
    ref.onDispose(() {
      _undoTimer?.cancel();
      _committedDeleteIds.clear();
    });
    return repo.watchAll().map((items) {
      _allItems = items;
      // Clean up committedDeleteIds once the stream no longer contains them.
      _committedDeleteIds.removeWhere(
        (id) => !items.any((i) => i.id == id),
      );
      return _buildState();
    });
  }

  ShoppingListState _buildState() {
    final hidden = <int>{..._committedDeleteIds};
    if (_pendingDelete != null) hidden.add(_pendingDelete!.id);
    final visible = hidden.isEmpty
        ? _allItems
        : _allItems.where((i) => !hidden.contains(i.id)).toList();
    return ShoppingListState.data(
      items: visible,
      pendingDelete: _pendingDelete,
    );
  }

  void setEffectListener(ShoppingListEffectListener? listener) {
    _effectListener = listener;
  }

  Future<void> deleteItem(int id) async {
    if (_pendingDelete != null) {
      _undoTimer?.cancel();
      _undoTimer = null;
      await _commitDelete(_pendingDelete!.id);
    }
    final item = _allItems.firstWhere(
      (i) => i.id == id,
      orElse: () => throw StateError('item $id not in list'),
    );
    _pendingDelete = item;
    _undoTimer = Timer(kUndoWindow, () async {
      final pending = _pendingDelete;
      if (pending == null) return;
      await _commitDelete(pending.id);
    });
  }

  Future<void> undoDelete() async {
    _undoTimer?.cancel();
    _undoTimer = null;
    _pendingDelete = null;
  }

  Future<bool> _commitDelete(int id) async {
    _committedDeleteIds.add(id);
    try {
      await ref.read(shoppingListRepositoryProvider).delete(id);
      if (_pendingDelete?.id == id) _pendingDelete = null;
      return true;
    } catch (error) {
      _committedDeleteIds.remove(id);
      if (_pendingDelete?.id == id) _pendingDelete = null;
      _effectListener?.call(ShoppingListDeleteFailedEffect(error));
      return false;
    }
  }
}
```

```dart
// lib/features/shopping_list/shopping_list_item_controller.dart
import 'package:riverpod_annotation/riverpod_annotation.dart';

import '../../app/providers/repository_providers.dart';
import '../../data/models/account.dart';
import '../../data/models/category.dart';
import '../../data/models/shopping_list_item.dart';
import '../../data/models/transaction.dart';
import '../transactions/keypad_state.dart';
import 'shopping_list_item_state.dart';

part 'shopping_list_item_controller.g.dart';

typedef ShoppingListItemEffectListener = void Function(ShoppingListItemEffect);

sealed class ShoppingListItemEffect { const ShoppingListItemEffect(); }
final class ShoppingListItemSavedEffect extends ShoppingListItemEffect {
  const ShoppingListItemSavedEffect();
}
final class ShoppingListItemConvertedEffect extends ShoppingListItemEffect {
  const ShoppingListItemConvertedEffect(this.transaction);
  final Transaction transaction;
}
final class ShoppingListItemDeletedEffect extends ShoppingListItemEffect {
  const ShoppingListItemDeletedEffect();
}
final class ShoppingListItemSaveFailedEffect extends ShoppingListItemEffect {
  const ShoppingListItemSaveFailedEffect(this.error);
  final Object error;
}

@Riverpod(dependencies: [
  shoppingListRepository,
  transactionRepository,
  categoryRepository,
  accountRepository,
])
class ShoppingListItemController extends _$ShoppingListItemController {
  KeypadState _keypad = const KeypadState.initial();
  ShoppingListItemEffectListener? _effectListener;
  ShoppingListItem? _originalItem;

  // Family parameter: null = new item, non-null = edit existing
  @override
  Future<ShoppingListItemState> build(int? itemId) async {
    if (itemId == null) {
      return const ShoppingListItemState.data(
        selectedCategory: null,
        selectedAccount: null,
        memo: '',
        amountMinorUnits: 0,
        isSaving: false,
        isConverting: false,
        isDeleting: false,
        showAmountHint: false,
      );
    }
    final item = await ref.read(shoppingListRepositoryProvider).getById(itemId);
    if (item == null) return const ShoppingListItemState.notFound();
    _originalItem = item;
    // Resolve account for currency inference
    final account = await ref
        .read(accountRepositoryProvider)
        .getById(item.accountId);
    final category = await ref
        .read(categoryRepositoryProvider)
        .getById(item.categoryId);
    if (item.draftAmountMinorUnits != null) {
      _keypad = _keypadFromAmount(item.draftAmountMinorUnits!,
          decimals: account?.currency.decimals ?? 2);
    }
    return ShoppingListItemState.data(
      selectedCategory: category,
      selectedAccount: account,
      memo: item.memo ?? '',
      amountMinorUnits: item.draftAmountMinorUnits ?? 0,
      isSaving: false,
      isConverting: false,
      isDeleting: false,
      showAmountHint: false,
    );
  }

  KeypadState get keypadSnapshot => _keypad;

  void setEffectListener(ShoppingListItemEffectListener? listener) {
    _effectListener = listener;
  }

  void setCategory(Category? category) {
    final s = state.valueOrNull;
    if (s is! ShoppingListItemData) return;
    state = AsyncData(s.copyWith(selectedCategory: category));
  }

  void setAccount(Account? account) {
    final s = state.valueOrNull;
    if (s is! ShoppingListItemData) return;
    // Re-infer currency from new account; draft currency is overwritten.
    state = AsyncData(s.copyWith(selectedAccount: account));
  }

  void setMemo(String memo) {
    final s = state.valueOrNull;
    if (s is! ShoppingListItemData) return;
    state = AsyncData(s.copyWith(memo: memo));
  }

  void appendDigit(int digit) => _mutateKeypad(
    _keypad.appendDigit(digit),
  );

  void appendDecimal() => _mutateKeypad(_keypad.appendDecimal());

  void backspace() => _mutateKeypad(_keypad.backspace());

  void clearAmount() => _mutateKeypad(const KeypadState.initial());

  void applyOperator(CalcOperator op) =>
      _mutateKeypad(_keypad.applyOperator(op));

  void _mutateKeypad(KeypadState next) {
    _keypad = next;
    final s = state.valueOrNull;
    if (s is! ShoppingListItemData) return;
    final amount = next.resolvedMinorUnits(
      decimals: s.selectedAccount?.currency.decimals ?? 2,
    );
    state = AsyncData(s.copyWith(
      amountMinorUnits: amount,
      keypadRevision: s.keypadRevision + 1,
    ));
  }

  bool get canSaveDraft {
    final s = state.valueOrNull;
    if (s is! ShoppingListItemData) return false;
    return s.selectedCategory != null &&
        !s.selectedCategory!.isArchived &&
        s.selectedAccount != null &&
        !s.selectedAccount!.isArchived &&
        !s.isSaving &&
        !s.isConverting;
  }

  bool get canSaveToTransaction {
    final s = state.valueOrNull;
    if (s is! ShoppingListItemData) return false;
    return s.selectedCategory != null &&
        !s.selectedCategory!.isArchived &&
        s.selectedAccount != null &&
        !s.selectedAccount!.isArchived &&
        s.amountMinorUnits > 0 &&
        !s.isSaving &&
        !s.isConverting;
  }

  Future<void> saveDraft() async {
    if (!canSaveDraft) return;
    final s = state.value as ShoppingListItemData;
    state = AsyncData(s.copyWith(isSaving: true));
    try {
      final itemId = _originalItem?.id ?? 0;
      final account = s.selectedAccount!;
      final currency = account.currency.code;
      final item = ShoppingListItem(
        id: itemId,
        categoryId: s.selectedCategory!.id,
        accountId: account.id,
        memo: s.memo.isEmpty ? null : s.memo,
        draftAmountMinorUnits: s.amountMinorUnits > 0 ? s.amountMinorUnits : null,
        draftCurrencyCode: s.amountMinorUnits > 0 ? currency : null,
        createdAt: _originalItem?.createdAt ?? DateTime.now(),
        updatedAt: DateTime.now(),
      );
      final saved = itemId == 0
          ? await ref.read(shoppingListRepositoryProvider).insert(item)
          : await ref.read(shoppingListRepositoryProvider).update(item);
      _originalItem = saved;
      state = AsyncData(s.copyWith(isSaving: false));
      _effectListener?.call(const ShoppingListItemSavedEffect());
    } catch (e) {
      state = AsyncData(s.copyWith(isSaving: false));
      _effectListener?.call(ShoppingListItemSaveFailedEffect(e));
    }
  }

  Future<void> saveToTransaction() async {
    final s = state.valueOrNull;
    if (s is! ShoppingListItemData) return;
    if (s.amountMinorUnits <= 0) {
      state = AsyncData(s.copyWith(showAmountHint: true));
      return;
    }
    if (!canSaveToTransaction) return;
    state = AsyncData(s.copyWith(isConverting: true));
    try {
      final account = s.selectedAccount!;
      final tx = Transaction(
        id: 0,
        amountMinorUnits: s.amountMinorUnits,
        currency: account.currency,
        categoryId: s.selectedCategory!.id,
        accountId: account.id,
        date: DateTime.now(),
        memo: s.memo.isEmpty ? null : s.memo,
        createdAt: DateTime.now(),
        updatedAt: DateTime.now(),
      );
      final saved = await ref.read(transactionRepositoryProvider).save(tx);
      // Only delete draft AFTER transaction saved successfully.
      final draftId = _originalItem?.id;
      if (draftId != null) {
        await ref.read(shoppingListRepositoryProvider).delete(draftId);
      }
      _effectListener?.call(ShoppingListItemConvertedEffect(saved));
    } catch (e) {
      // Do NOT delete draft on failure.
      state = AsyncData(s.copyWith(isConverting: false));
      _effectListener?.call(ShoppingListItemSaveFailedEffect(e));
    }
  }

  Future<void> delete() async {
    final s = state.valueOrNull;
    if (s is! ShoppingListItemData) return;
    final id = _originalItem?.id;
    if (id == null) return;
    state = AsyncData(s.copyWith(isDeleting: true));
    try {
      await ref.read(shoppingListRepositoryProvider).delete(id);
      _effectListener?.call(const ShoppingListItemDeletedEffect());
    } catch (e) {
      state = AsyncData(s.copyWith(isDeleting: false));
      _effectListener?.call(ShoppingListItemSaveFailedEffect(e));
    }
  }

  KeypadState _keypadFromAmount(int minorUnits, {required int decimals}) {
    // Rebuild keypad representation from stored amount.
    final factor = _pow10(decimals);
    final intPart = minorUnits ~/ factor;
    final fracPart = minorUnits % factor;
    if (decimals == 0 || fracPart == 0) {
      return KeypadState(
        display: '$intPart',
        pendingOperator: null,
        lhsMinorUnits: null,
      );
    }
    final fracStr = fracPart.toString().padLeft(decimals, '0');
    return KeypadState(
      display: '$intPart.$fracStr',
      pendingOperator: null,
      lhsMinorUnits: null,
    );
  }

  int _pow10(int n) {
    var result = 1;
    for (var i = 0; i < n; i++) result *= 10;
    return result;
  }
}
```

```dart
  ConsumerState<ShoppingListScreen> createState() =>

class _ShoppingListScreenState extends ConsumerState<ShoppingListScreen> {
  ShoppingListData? _lastPendingShown;

  @override
  void initState() {
    super.initState();
    ref
        .read(shoppingListControllerProvider.notifier)
        .setEffectListener(_onEffect);
  }

  @override
  void dispose() {
    ref
        .read(shoppingListControllerProvider.notifier)
        .setEffectListener(null);
    super.dispose();
  }

  void _onEffect(ShoppingListEffect effect) {
    if (!mounted) return;
    final l10n = AppLocalizations.of(context);
    switch (effect) {
      case ShoppingListDeleteFailedEffect():
        ScaffoldMessenger.of(context)
          ..clearSnackBars()
          ..showSnackBar(
            SnackBar(content: Text(l10n.errorSnackbarGeneric)),
          );
    }
  }

  @override
  Widget build(BuildContext context) {
    final l10n = AppLocalizations.of(context);
    final state = ref.watch(shoppingListControllerProvider);

    ref.listen(shoppingListControllerProvider, (_, next) {
      if (next is AsyncData<ShoppingListState>) {
        final value = next.value;
        if (value is ShoppingListData && value.pendingDelete != null) {
          if (_lastPendingShown?.pendingDelete?.id !=
              value.pendingDelete?.id) {
            _lastPendingShown = value;
            _showUndoSnackbar(context, l10n);
          }
        }
      }
    });

    return Scaffold(
      appBar: AppBar(title: Text(l10n.shoppingListScreenTitle)),
      floatingActionButton: FloatingActionButton(
        heroTag: 'shopping_list_fab',
        onPressed: () => context.push('/accounts/shopping-list/new'),
        child: const Icon(Icons.add),
      ),
      body: SlidableAutoCloseBehavior(
        child: switch (state) {
          AsyncData<ShoppingListState>(
            value: ShoppingListData(items: []),
          ) =>
            Center(
              child: Text(
                l10n.shoppingListEmptyState,
                style: Theme.of(context).textTheme.bodyMedium,
              ),
            ),
          AsyncData<ShoppingListState>(
            value: ShoppingListData(:final items),
          ) =>
            ListView.builder(
              itemCount: items.length,
              itemBuilder: (context, index) {
                final item = items[index];
                return Slidable(
                  key: ValueKey<int>(item.id),
                  endActionPane: ActionPane(
                    motion: const DrawerMotion(),
                    children: [
                      SlidableAction(
                        onPressed: (_) => ref
                            .read(shoppingListControllerProvider.notifier)
                            .deleteItem(item.id),
                        backgroundColor:
                            Theme.of(context).colorScheme.errorContainer,
                        foregroundColor:
                            Theme.of(context).colorScheme.onErrorContainer,
                        icon: Icons.delete_outline,
                        label: l10n.commonDelete,
                      ),
                    ],
                  ),
                  child: ListTile(
                    title: Text(item.memo ?? ''),
                    onTap: () =>
                        context.push('/accounts/shopping-list/${item.id}'),
                  ),
                );
              },
            ),
          _ => const Center(child: CircularProgressIndicator()),
        },
      ),
    );
  }

  void _showUndoSnackbar(BuildContext context, AppLocalizations l10n) {
    ScaffoldMessenger.of(context)
      ?..clearSnackBars()
      ..showSnackBar(
        SnackBar(
          content: Text(l10n.shoppingListDeleteUndoSnackbar),
          duration: kUndoWindow,
          action: SnackBarAction(
            label: l10n.commonUndo,
            onPressed: () => ref
                .read(shoppingListControllerProvider.notifier)
                .undoDelete(),
          ),
        ),
      );
```

```dart
  @override
  void initState() {

    WidgetsBinding.instance.addPostFrameCallback((_) {

          .read(shoppingListItemControllerProvider(widget.itemId).notifier)

      case ShoppingListItemConvertedEffect():
        context.pop();
      case ShoppingListItemDeletedEffect():
        context.pop();
      case ShoppingListItemSaveFailedEffect():

          SnackBar(content: Text(l10n.shoppingListConvertFailedSnackbar)),

  Widget build(BuildContext context) {
    final l10n = AppLocalizations.of(context);

      shoppingListItemControllerProvider(widget.itemId),

      shoppingListItemControllerProvider(widget.itemId).notifier,

      appBar: AppBar(

              tooltip: l10n.commonDelete,

                          !(state.value as ShoppingListItemData).isDeleting

                      : null,

      body: SafeArea(
        child: switch (state) {
          AsyncLoading() => const Center(child: CircularProgressIndicator()),
          AsyncData(value: ShoppingListItemNotFound()) => Center(
            child: Text(l10n.shoppingListItemNotFound),
          ),
          AsyncData(value: ShoppingListItemData() && final s) =>
            _buildForm(context, l10n, s, notifier),

                          child: Icon(Icons.warning_amber,

                              size: 16),
                        ),
                    ],
                  ),
                  Stack(
                    children: [

                        account: state.selectedAccount,

                        },

                      if (archivedAccount)

                          top: 0,
                          right: 0,
                          child: Icon(Icons.warning_amber,
                              color: Theme.of(context).colorScheme.error,
                              size: 16),
```
